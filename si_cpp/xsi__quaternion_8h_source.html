526 SICPPSDK_INLINE CQuaternion& CQuaternion::Mul
527 (
528  const CQuaternion &in_Quat1,
529  const CQuaternion &in_Quat2
530 )
531 {
532  double dW, dX, dY, dZ;
533 
534  dW = in_Quat1.m_dW * in_Quat2.m_dW - in_Quat1.m_dX * in_Quat2.m_dX -
535  in_Quat1.m_dY * in_Quat2.m_dY - in_Quat1.m_dZ * in_Quat2.m_dZ ;
536 
537  dX = in_Quat1.m_dW * in_Quat2.m_dX + in_Quat2.m_dW * in_Quat1.m_dX +
538  in_Quat1.m_dY * in_Quat2.m_dZ - in_Quat2.m_dY * in_Quat1.m_dZ ;
539 
540  dY = in_Quat1.m_dW * in_Quat2.m_dY + in_Quat2.m_dW * in_Quat1.m_dY -
541  in_Quat1.m_dX * in_Quat2.m_dZ + in_Quat2.m_dX * in_Quat1.m_dZ ;
542 
543  dZ = in_Quat1.m_dW * in_Quat2.m_dZ + in_Quat2.m_dW * in_Quat1.m_dZ +
544  in_Quat1.m_dX * in_Quat2.m_dY - in_Quat2.m_dX * in_Quat1.m_dY ;
545 
546  m_dW = dW; m_dX = dX; m_dY = dY; m_dZ = dZ;
547 
548  return *this;
549 }
