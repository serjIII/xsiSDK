622 void CRotationf::SetRepresentation( const RotationRep in_rep, const RotationOrder in_order )
623 {
624  if ( m_rotRep != in_rep )
625  {
626  switch( in_rep )
627  {
628  case siQuaternionRot:
629  {
630  CQuaternionf l_quat;
631  switch ( m_rotRep )
632  {
633  case siEulerRot:
634  {
635  CVector3f l_euler( m_X, m_Y, m_Z );
636  XSI::MATH::EulerToQuat( l_euler, l_quat, m_rotOrder );
637  }
638  break;
639  case siAxisAngleRot:
640  {
641  CVector3f l_axis( m_X, m_Y, m_Z );
642  if ( l_axis.GetLength() > MicroEPS )
643  {
644  l_axis.NormalizeInPlace();
645  AxisAngleToQuat( l_axis, m_Angle, l_quat );
646  }
647  }
648  break;
649  }
650  *this = l_quat;
651  }
652  break;
653 
654  case siEulerRot:
655  {
656  CVector3f l_euler;
657  switch ( m_rotRep )
658  {
659  case siQuaternionRot:
660  {
661  CQuaternionf l_quat( m_W, m_X, m_Y, m_Z );
662  l_quat.NormalizeInPlace();
663  QuatToEuler( l_quat, l_euler, in_order );
664  }
665  break;
666  case siAxisAngleRot:
667  {
668  CQuaternionf l_quat;
669  CVector3f l_axis( m_X, m_Y, m_Z );
670  if ( l_axis.GetLength() > MicroEPS )
671  {
672  l_axis.NormalizeInPlace();
673  AxisAngleToQuat( l_axis, m_Angle, l_quat );
674  }
675  QuatToEuler( l_quat, l_euler, in_order );
676  }
677  break;
678  }
679  Set( l_euler, in_order );
680  }
681  break;
682  case siAxisAngleRot:
683  {
684  float l_angle = 0;
685  CVector3f l_axis;
686  switch ( m_rotRep )
687  {
688  case siQuaternionRot:
689  {
690  CQuaternionf l_quat( m_W, m_X, m_Y, m_Z );
691  l_quat.NormalizeInPlace();
692  QuatToAxisAngle( l_quat, l_axis, l_angle );
693  }
694  break;
695 
696  case siEulerRot:
697  {
698  CQuaternionf l_quat;
699  CVector3f l_euler( m_X, m_Y, m_Z );
700  XSI::MATH::EulerToQuat( l_euler, l_quat, m_rotOrder );
701  QuatToAxisAngle( l_quat, l_axis, l_angle );
702  }
703  break;
704  }
705 
706  if ( l_axis.IsNull() )
707  {
708  l_axis.Set(1.0, 0.0, 0.0); // If axis is undefined set it to a valid axis
709  }
710  Set( l_axis, l_angle );
711  }
712  break;
713  }
714  }

748 SICPPSDK_INLINE bool CRotationf::EpsilonEquals( const CRotationf& in_rot, const float in_fEpsilon ) const
749 {
750  CRotationf r1( *this );
751  CRotationf r2( in_rot );
752  if ( r1.GetRepresentation() != r2.GetRepresentation() )
753  {
754  r2.SetRepresentation( r1.GetRepresentation() );
755  }
756  switch (r1.GetRepresentation())
757  {
758  case siQuaternionRot:
759  {
760  return r1.GetQuaternion().EpsilonEquals( r2.GetQuaternion(), in_fEpsilon );
761  }
762  case siEulerRot:
763  {
764  return ( r1.GetOrder() == r2.GetOrder() && r1.GetXYZAngles().EpsilonEquals( r2.GetXYZAngles(), in_fEpsilon ) );
765  }
766  case siAxisAngleRot:
767  {
768  float a1, a2;
769  CVector3f v1 = r1.GetAxisAngle( a1 );
770  CVector3f v2 = r2.GetAxisAngle( a2 );
771  return ( fabs( a1 - a2 ) < in_fEpsilon && v1.EpsilonEquals( v2, in_fEpsilon ) );
772  }
773  default:
774  break;
775  }
776  return false;
777 }

800 SICPPSDK_INLINE void CRotationf::QuatToMat( const CQuaternionf& in_quat, CMatrix3f& out_mat )
801 {
802  float d2X, d2Y, d2Z, d2X2, d2Y2, d2Z2, d2WX, d2WY, d2WZ, d2XY, d2XZ, d2YZ;
803 
804  d2X = (float)2.0 * in_quat.GetX();
805  d2Y = (float)2.0 * in_quat.GetY();
806  d2Z = (float)2.0 * in_quat.GetZ();
807 
808  d2X2 = d2X * in_quat.GetX();
809  d2Y2 = d2Y * in_quat.GetY();
810  d2Z2 = d2Z * in_quat.GetZ();
811 
812  d2WX = d2X * in_quat.GetW();
813  d2WY = d2Y * in_quat.GetW();
814  d2WZ = d2Z * in_quat.GetW();
815  d2XY = d2X * in_quat.GetY();
816  d2YZ = d2Y * in_quat.GetZ();
817  d2XZ = d2Z * in_quat.GetX();
818 
819  out_mat.SetValue( 0, 0, (float)1.0 - d2Y2 - d2Z2 );
