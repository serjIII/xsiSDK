451 )

724 SICPPSDK_INLINE CQuaternionf& CQuaternionf::NormalizeInPlace()
725 {

728  if( fabs( rLen - 1.0f ) >= PicoEPS )
729  {

754 SICPPSDK_INLINE CQuaternionf& CQuaternionf::Slerp
755 (
756  const CQuaternionf& in_quatStart,
757  const CQuaternionf& in_quatEnd,
758  const float in_fU
759 )
760 {
761  float fDot, fScale1, fScale2;
762  bool bNegate = false;
763 
764  fDot = in_quatStart.Dot( in_quatEnd );
765  if ( fDot < 0.0 )
766  {
767  fDot = -fDot;
768  bNegate = true;
769  }
770 
771  if( fDot >= ( 1.0f - MicroEPS ) )
772  {
773  fScale1 = 1.0f - in_fU;
774  fScale2 = in_fU;
775  }
776  else
777  {
778  float fAngle, fFactor;
779 
780  LimitValue( fDot, -1.0, 1.0 );
781  fAngle = acos( fDot );
782  fFactor = 1.0f / sin( fAngle );
783 
784  fScale1 = sin( (1.0f - in_fU) * fAngle ) * fFactor;
785  fScale2 = sin( in_fU * fAngle ) * fFactor;
786  }
787 
788  if( bNegate == true )
789  fScale2 = -fScale2;
790 
791  m_W = fScale1 * in_quatStart.m_W + fScale2 * in_quatEnd.m_W;
792  m_X = fScale1 * in_quatStart.m_X + fScale2 * in_quatEnd.m_X;
793  m_Y = fScale1 * in_quatStart.m_Y + fScale2 * in_quatEnd.m_Y;
794  m_Z = fScale1 * in_quatStart.m_Z + fScale2 * in_quatEnd.m_Z;
795 
796  return( *this );
797 }
