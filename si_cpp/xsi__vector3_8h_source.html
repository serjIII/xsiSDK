775 SICPPSDK_INLINE CVector3& CVector3::Cross
776 (
777  const CVector3& in_vec3A,
778  const CVector3& in_vec3B
779 )
780 {
781  double dX, dY, dZ;
782 
783  dX = in_vec3A.GetY() * in_vec3B.GetZ() -
784  in_vec3A.GetZ() * in_vec3B.GetY();
785  dY = -in_vec3A.GetX() * in_vec3B.GetZ() +
786  in_vec3A.GetZ() * in_vec3B.GetX();
787  dZ = in_vec3A.GetX() * in_vec3B.GetY() -
788  in_vec3A.GetY() * in_vec3B.GetX();
789 
790  return( Set( dX, dY, dZ ) );
