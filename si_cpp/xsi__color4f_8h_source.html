357 SICPPSDK_INLINE void CColor4f::Set( float in_Arg1, float in_Arg2, float in_Arg3, float in_A, Type in_eColorType )
358 {
359  m_RGBA[A] = in_A;
360 
361  switch( in_eColorType )
362  {
363  case siRGBA:
364  {
365  m_RGBA[R] = in_Arg1;
366  m_RGBA[G] = in_Arg2;
367  m_RGBA[B] = in_Arg3;
368  break;
369  }
370  case siHSVA:
371  {
372  if (in_Arg2 == 0.0)
373  {
374  m_RGBA[R] = in_Arg3;
375  m_RGBA[G] = in_Arg3;
376  m_RGBA[B] = in_Arg3;
377  }
378  else
379  {
380  in_Arg1 *= (float)( 360.0/60.0 );
381 
382  ULONG l_ulHueI = (ULONG)RoundDown(in_Arg1);
383  float l_fHueF = in_Arg1 - l_ulHueI;
384 
385  float l_fP = in_Arg3 * (1.0f - in_Arg2);
386  float l_fQ = in_Arg3 * (1.0f - (in_Arg2*l_fHueF));
387  float l_fT = in_Arg3 * (1.0f - (in_Arg2*(1.0f-l_fHueF)));
388 
389  switch (l_ulHueI)
390  {
391  case 0 :
392  case 6 : m_RGBA[R] = in_Arg3; m_RGBA[G] = l_fT; m_RGBA[B] = l_fP;
393  break;
394  case 1 : m_RGBA[R] = l_fQ; m_RGBA[G] = in_Arg3; m_RGBA[B] = l_fP;
395  break;
396  case 2 : m_RGBA[R] = l_fP; m_RGBA[G] = in_Arg3; m_RGBA[B] = l_fT;
397  break;
398  case 3 : m_RGBA[R] = l_fP; m_RGBA[G] = l_fQ; m_RGBA[B] = in_Arg3;
399  break;
400  case 4 : m_RGBA[R] = l_fT; m_RGBA[G] = l_fP; m_RGBA[B] = in_Arg3;
401  break;
402  case 5 : m_RGBA[R] = in_Arg3; m_RGBA[G] = l_fP; m_RGBA[B] = l_fQ;
403  break;
404  }
405  }
406 
407  break;
408  }
409  case siHLSA:
410  {
411  float l_fMax;
412 
413  if (in_Arg2 <= 0.5)
414  l_fMax = in_Arg2 * (1.0f + in_Arg3);
415  else
416  l_fMax = in_Arg2 + in_Arg3 - (in_Arg2 * in_Arg3);
417 
418  if (l_fMax == 0.0f)
419  {
420  m_RGBA[R] = m_RGBA[G] = m_RGBA[B] = 0.0f;
421  }
422  else
423  {
424  in_Arg1 *= (float)( 360.0/60.0 );
425 
426  ULONG l_ulHueI = (ULONG)RoundDown(in_Arg1);
427  float l_fHueF = in_Arg1 - l_ulHueI;
428 
429  float l_fMin = 2.0f*in_Arg2 - l_fMax;
430  float l_fQ = l_fMax - ((l_fMax-l_fMin)*l_fHueF);
431  float l_fT = l_fMin + ((l_fMax-l_fMin)*l_fHueF);
432 
433  switch (l_ulHueI)
434  {
435  case 0 :
436  case 6 : m_RGBA[R] = l_fMax; m_RGBA[G] = l_fT; m_RGBA[B] = l_fMin;
437  break;
438  case 1 : m_RGBA[R] = l_fQ; m_RGBA[G] = l_fMax; m_RGBA[B] = l_fMin;
439  break;
440  case 2 : m_RGBA[R] = l_fMin; m_RGBA[G] = l_fMax; m_RGBA[B] = l_fT;
441  break;
442  case 3 : m_RGBA[R] = l_fMin; m_RGBA[G] = l_fQ; m_RGBA[B] = l_fMax;
443  break;
444  case 4 : m_RGBA[R] = l_fT; m_RGBA[G] = l_fMin; m_RGBA[B] = l_fMax;
445  break;
446  case 5 : m_RGBA[R] = l_fMax; m_RGBA[G] = l_fMin; m_RGBA[B] = l_fQ;
447  break;
448  }
449  }
450  break;
451  }
452  default:
453  assert( false );

468  float l_fDelta = l_fMax - l_fMin;

471

484  else

489  out_fHue = 2+(m_RGBA[B]-m_RGBA[R])/l_fDelta;

492 
493  out_fHue *= 60;
494 
495  if (out_fHue < 0)
